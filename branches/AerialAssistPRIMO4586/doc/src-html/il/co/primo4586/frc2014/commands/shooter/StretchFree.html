006package il.co.primo4586.frc2014.commands.shooter;
007
008import edu.wpi.first.wpilibj.DriverStationLCD;
009import edu.wpi.first.wpilibj.Joystick;
010import il.co.primo4586.frc2014.commands.CommandBase;
011/**
012 *
013 * @author mor meitar idan
014 */
015public class StretchFree extends CommandBase {
016    /**
017     *  Victor stretcher;
018        Talon releaser;
019        AnalogChannel cycleCounter;
020        DigitalInput stretcherStart;
021        DigitalInput stretcherEnd;
022        DigitalInput releaserLock;
023        double stretchCycles, strechDistance;
024        long primeCountCycles; // prime cycles count situations
025        long newCountCycles; // the cycles count since the prime situation 
026     */
027      
028    private Joystick operatorStick;
029    private double direction;
030    
031    public StretchFree()         
032    {
033        requires(shooter);
034        // Use requires() here to declare subsystem dependencies
035        // eg. requires(chassis);
036    }
037
038    // Called just before this Command runs the first time
039    protected void initialize() 
040    {
041        operatorStick = oi.operatorStick;
042        shooter.initCount();
043
044    }
045
046    // Called repeatedly when this Command is scheduled to run
047    protected void execute() 
048    {
049            direction = operatorStick.getRawAxis(1) / Math.abs(operatorStick.getRawAxis(1));  // check which port is it
050        
051        if ( (direction > 0 &&  !shooter.getEndMicro()) ||  (direction < 0 &&  !shooter.getStartMicro()) )
052        {
053            shooter.stretch(direction/2);
054            DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser2, 1, "stretch cycles:" + shooter.getCount());
055            DriverStationLCD.getInstance().updateLCD();
056        }
057    }
058
059    // Make this return true when this Command no longer needs to run execute()
060    protected boolean isFinished() 
061    {
062        return false;
063    }
064
065    // Called once after isFinished returns true
066    protected void end() 
067    {
068        
069    }
070
071    // Called when another command which requires one or more of the same
072    // subsystems is scheduled to run
073    protected void interrupted() 
074    {
075        
076    }
077}
