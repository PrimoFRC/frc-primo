002package il.co.primo4586.frc2014;
003
004
005import edu.wpi.first.wpilibj.Joystick;
006import edu.wpi.first.wpilibj.buttons.Button;
007import edu.wpi.first.wpilibj.buttons.DigitalIOButton;
008import edu.wpi.first.wpilibj.buttons.JoystickButton;
009import il.co.primo4586.frc2014.commands.Aim;
010import il.co.primo4586.frc2014.commands.shooter.*;
011import il.co.primo4586.frc2014.commands.collector.*;
012
013/**
014 * This class is the glue that binds the controls on the physical operator
015 * interface to the commands and command groups that allow control of the robot.
016 */
017public class OI {
018    //// CREATING BUTTONS
019    // One type of button is a joystick button which is any button on a joystick.
020    // You create one by telling it which joystick it's on and which button
021    // number it is.
022    // Joystick stick = new Joystick(port);
023    // Button button = new JoystickButton(stick, buttonNumber);
024
025    // Another type of button you can create is a DigitalIOButton, which is
026    // a button or switch hooked up to the cypress module. These are useful if
027    // you want to build a customized operator interface.
028    // Button button = new DigitalIOButton(1);
029
030    // There are a few additional built in buttons you can use. Additionally,
031    // by subclassing Button you can create custom triggers and bind those to
032    // commands the same as any other Button.
033
034    //// TRIGGERING COMMANDS WITH BUTTONS
035    // Once you have a button, it's trivial to bind it to a button in one of
036    // three ways:
037
038    // Start the command when the button is pressed and let it run the command
039    // until it is finished as determined by it's isFinished method.
040    // button.whenPressed(new ExampleCommand());
041
042    // Run the command while the button is being held down and interrupt it once
043    // the button is released.
044    // button.whileHeld(new ExampleCommand());
045
046    // Start the command when the button is released  and let it run the command
047    // until it is finished as determined by it's isFinished method.
048    // button.whenReleased(new ExampleCommand());
049
050    public Joystick drivingStick;
051    public Joystick operatorStick;
052
053    public JoystickButton manualLowerCollectorArm, manualRaiseCollectorArm, collectorToBottom, collectorToTop, collectorToMiddle ;
054    public JoystickButton shoot , aim , trussThrow , initStretcher;
055
056
057    public OI()
058    {
059                drivingStick = new Joystick(1);
060        operatorStick = new Joystick(2);
061
062
063                collectorToBottom = new JoystickButton(operatorStick, 1);
064                collectorToTop = new JoystickButton(operatorStick, 2);
065                collectorToMiddle = new JoystickButton(operatorStick, 3);
066
067                collectorToBottom.whenPressed(new MoveArmBottom());
068                collectorToTop.whenPressed(new MoveArmTop());
069                collectorToMiddle.whenPressed(new MoveArmMiddle());
070
071
072                aim = new JoystickButton(operatorStick, 7);
073                shoot = new JoystickButton(operatorStick, 8);
074                trussThrow = new JoystickButton(operatorStick, 6);
075                initStretcher = new JoystickButton(operatorStick, 5);
076
077                aim.whenPressed(new Aim());
078                shoot.whenPressed(new Shoot());
079                trussThrow.whenPressed(new ThrowOverTruss());
080                initStretcher.whenPressed(new InitStretcher());
081
082    }
083
084}
