014public class Driver extends Subsystem {
015    // Put methods for controlling this subsystem
016    // here. Call these from Commands.
017    public Jaguar frontLeft;
018    public Jaguar frontRight;
019    public Jaguar backLeft;
020    public Jaguar backRight;
021    
022    public RobotDrive driver;
023    
024    public Driver()
025    {
026        frontLeft = RobotMap.frontLeft;
027        frontRight = RobotMap.frontRight;
028        backLeft = RobotMap.backLeft;
029        backRight = RobotMap.backRight;
030        
031        driver = new RobotDrive(frontLeft, backLeft, frontRight, backRight);
032    }
033     
034    public void arcadeDrive(Joystick stick)
035    {
036        //driver.arcadeDrive(stick);
037        driver.arcadeDrive(-stick.getY()*SmartDashboard.getNumber("drive move speed: "), -stick.getX()*SmartDashboard.getNumber("drive move speed: "));
038    }
039    
040    public void moveForward()
041    {
042        frontLeft.set(-0.5*SmartDashboard.getNumber("drive move speed: "));
043        frontRight.set(-0.5*SmartDashboard.getNumber("drive move speed: "));
044        backLeft.set(-0.5*SmartDashboard.getNumber("drive move speed: "));
045        backRight.set(-0.5*SmartDashboard.getNumber("drive move speed: "));
046    }
047    
048    public void stop()
049    {
050        frontLeft.set(0);
051        frontRight.set(0);
052        backLeft.set(0);
053        backRight.set(0);
054    }
