001package edu.wpi.first.wpilibj.templates;

003import edu.wpi.first.wpilibj.Counter;
004import edu.wpi.first.wpilibj.DigitalInput;
005import edu.wpi.first.wpilibj.Jaguar;
006import edu.wpi.first.wpilibj.Talon;

008/**
009 * The RobotMap is a mapping from the ports sensors and actuators are wired into
010 * to a variable name. This provides flexibility changing wiring, makes checking
011 * the wiring easier and significantly reduces the number of magic numbers
012 * floating around.
013 */
014public class RobotMap {
015    // For example to map the left and right motors, you could define the
016    // following variables to use with your drivetrain subsystem.
017    // public static final int leftMotor = 1;
018    // public static final int rightMotor = 2;
019    
020    // If you are using multiple modules, make sure to define both the port
021    // number and the module. For example you with a rangefinder:
022    // public static final int rangefinderPort = 1;
023    // public static final int rangefinderModule = 1;
024    
025    
026    //microswitches
027    public static DigitalInput topLifter;
028    public static DigitalInput bottomLifter;
029    public static DigitalInput collection;
030    public static DigitalInput hanging;
031    
032    public static DigitalInput doorOpen;
033    public static DigitalInput doorClose;
034    
035    public static DigitalInput cycleSensor;
036    public static Counter cycleCounterPositive, cycleCounterNegative; 
037    
038    //speedControllers
039    public static Jaguar frontLeft;
040    public static Jaguar frontRight;
041    public static Jaguar backLeft;
042    public static Jaguar backRight;
043    
044    public static Talon doorMover;
045    public static Talon lifterMover;
046    
047    public static void init()
048    {
049        //DigitalInputs
050        topLifter = new DigitalInput(3);
051        bottomLifter = new DigitalInput(12);
052        
053        doorOpen = new DigitalInput(8);
054        doorClose = new DigitalInput(5);
055        
056        cycleSensor = new DigitalInput(10);
057        cycleCounterPositive = new Counter(cycleSensor);
058        cycleCounterNegative = new Counter(cycleSensor);
059                
060        //speedControllers
061        frontLeft = new Jaguar(1);
062        frontRight = new Jaguar(2);
063        backLeft = new Jaguar(4);
064        backRight = new Jaguar(3);
065        
066        doorMover = new Talon(6);
067        lifterMover = new Talon(5);
068    }
069}
