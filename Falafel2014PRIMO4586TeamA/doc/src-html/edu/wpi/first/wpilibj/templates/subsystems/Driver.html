030    {
031        //הפעולה מעניקה למקל נהיגה שליטה על תנועת הרובוט  
032        this.driver.arcadeDrive(joystick);

034    public void setSpeedXY(double x, double y)
035    {
036        //הפעולה מקבלת מהירויות x ו-y
037        //הפעולה מזיזה את הרובוט קדימה במהירות -y ומסובבות אותו עם כיוון השעון במהירות x
038        double left=Math.min(Math.max(x+y, -1), 1);
039        double right=left-2*x;
040        this.frontLeft.set(left);
041        this.backLeft.set(left);
042        this.frontRight.set(right);
043        //this.backRight.set(right);
044    }
